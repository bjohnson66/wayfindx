WayFindX Version 1.0
WayFindX is a handheld GPS receiver project developed by Avengers Assembly, consisting of Bradley Johnson and Abele Atresso. The project aims to create a portable navigation device that enhances outdoor navigation capabilities for users such as hikers, adventurers, and outdoor enthusiasts. WayFindX provides essential features including position and velocity tracking, storage of multiple locations, and distance calculation to selected stored positions.

114void get_serial_char(char* outputchar){
115 unsigned int c;
116 while (1){
117 /*
118 * Get received character from ringbuffer
119 * uart_getc() returns in the lower byte the received character and
120 * in the higher byte (bitmask) the last receive error
121 * UART_NO_DATA is returned when no data is available.
122 *
123 */
124 c = uart_getc();
125 if (!( c & UART_NO_DATA ))
126 {
127 /*
128 * new data available from UART
129 * check for Frame or Overrun error
130 */
131 #ifdef __DEBUG__
132 if ( c & UART_FRAME_ERROR )
133 {
134 /* Framing Error detected, i.e no stop bit detected */
135 #ifdef _DEBUG_
136 char* output = "NF Frame Error! ";
137 ds_print_string(output, 16, 0);
138 #endif
139 }
140 if ( c & UART_OVERRUN_ERROR )
141 {
142 /*
143 * Overrun, a character already present in the UART UDR register was
144 * not read by the interrupt handler before the next character arrived,
145 * one or more received characters have been dropped
146 */
147 char* output = " OR ER";
148 ds_print_string(output, 16, 0);
149 }
150 if ( c & UART_BUFFER_OVERFLOW )
151 {
152 /*
153 * We are not reading the receive buffer fast enough,
154 * one or more received character have been dropped
155 */
156 char* output = "OF ER";
157 ds_print_string(output, 5, 0);
158 }
159 #endif
160
161 /*
162 * send received character back
163 */
164 *outputchar = (unsigned char)c;
165 return;
166 }
167 }
